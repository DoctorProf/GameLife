    bool posInsideToWallArea(Vector2f start_pos, Vector2f end_pos, Vector2f pos)
    {
        Vector2f w = end_pos - start_pos;
        float dist_a = distance(start_pos, pos);
        float dist_b = distance(end_pos, pos);

        return length(w) <= sqrtf(powf(dist_a, 2) + powf(dist_b, 2));
    }
