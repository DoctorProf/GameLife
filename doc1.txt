## JSON — сохранение и загрузка настроек

### 1. **Базовые операции с JSON (nlohmann/json)**

```cpp
#include <nlohmann/json.hpp>
#include <fstream>
#include <vector>
#include <map>

using json = nlohmann::json;

// Создание JSON объекта
json settings;
settings["window"]["width"] = 800;
settings["window"]["height"] = 600;
settings["window"]["fullscreen"] = false;

settings["game"]["speed"] = 60;
settings["game"]["pause"] = true;
settings["game"]["grid_enabled"] = true;

// Массивы
settings["colors"]["background"] = {255, 255, 255, 255};
settings["colors"]["cell_alive"] = {255, 0, 0, 255};

// Вложенные объекты
json rules;
rules["classic"]["birth"] = {3};
rules["classic"]["survive"] = {2, 3};
rules["highlife"]["birth"] = {3, 6};
rules["highlife"]["survive"] = {2, 3};
settings["rules"] = rules;

// Сохранение в файл
void saveToFile(const json& data, const std::string& filename) {
    std::ofstream file(filename);
    if (file.is_open()) {
        file << data.dump(4); // 4 пробела для форматирования
    }
}

// Загрузка из файла
json loadFromFile(const std::string& filename) {
    std::ifstream file(filename);
    if (file.is_open()) {
        return json::parse(file);
    }
    return json{}; // пустой JSON если файл не найден
}
```

```cpp
// Сохраняем состояние всех виджетов в JSON
json saveWidgetsState(tgui::Gui& gui) {
    json state;
    
    // Чекбоксы
    if (auto cb = gui.get<tgui::CheckBox>("show_grid")) {
        state["show_grid"] = cb->isChecked();
    }
    if (auto cb = gui.get<tgui::CheckBox>("pause")) {
        state["pause"] = cb->isChecked();
    }
    
    // Слайдеры
    if (auto slider = gui.get<tgui::Slider>("speed")) {
        state["speed"] = slider->getValue();
    }
    
    // Поля ввода
    if (auto input = gui.get<tgui::EditBox>("cell_size")) {
        state["cell_size"] = input->getText().toInt();
    }
    if (auto input = gui.get<tgui::EditBox>("rule_name")) {
        state["rule_name"] = input->getText(); // TGUI::String конвертируется в std::string
    }
    
    // Комбобоксы
    if (auto combo = gui.get<tgui::ComboBox>("rule_select")) {
        state["selected_rule"] = combo->getSelectedItem(); // или getSelectedItemId()
        state["selected_index"] = combo->getSelectedItemIndex();
    }
    
    // Цвета
    if (auto panel = gui.get<tgui::Panel>("color_panel")) {
        auto color = panel->getRenderer()->getBackgroundColor();
        state["panel_color"] = {color.red, color.green, color.blue, color.alpha};
    }
    
    return state;
}

// Восстанавливаем состояние виджетов из JSON
void loadWidgetsState(tgui::Gui& gui, const json& state) {

    // Слайдеры
```

### 3. **Сохранение игровых структур**

```cpp
// Сохранение матрицы (поля игры)
json saveMatrix(const std::vector<std::vector<int>>& matrix) {
    json j;
    for (size_t y = 0; y < matrix.size(); ++y) {
        json row;
        for (size_t x = 0; x < matrix[y].size(); ++x) {
            row.push_back(matrix[y][x]);
        }
        j["data"].push_back(row);
    }
    j["height"] = matrix.size();
    j["width"] = matrix.empty() ? 0 : matrix[0].size();
    return j;
}

// Загрузка матрицы
std::vector<std::vector<int>> loadMatrix(const json& j) {
    std::vector<std::vector<int>> matrix;
    if (!j.contains("data")) return matrix;
    
    for (const auto& rowJson : j["data"]) {
        std::vector<int> row;
        for (int val : rowJson) {
            row.push_back(val);
        }
        matrix.push_back(row);
    }
    return matrix;
}

// Сохранение нескольких структур (например, избранные паттерны)
json savePatterns(const std::vector<std::pair<std::string, std::vector<std::vector<int>>>>& patterns) {
    json j;
    for (const auto& [name, pattern] : patterns) {
        j[name] = saveMatrix(pattern);
    }
    return j;
}
```

### 4. **Расширяемая система конфигурации**

```cpp
// Базовый класс для настраиваемых объектов
class IConfigurable {
public:
    virtual ~IConfigurable() = default;
    virtual json serialize() const = 0;
    virtual void deserialize(const json& j) = 0;
};

// Пример правила, которое можно сохранять
class GameRule : public IConfigurable {
private:
    std::string m_name;
    std::vector<int> m_birthConditions;
    std::vector<int> m_surviveConditions;
    int m_maxNeighbors = 8;
    
public:
    json serialize() const override {
        return {
            {"name", m_name},
            {"birth", m_birthConditions},
            {"survive", m_surviveConditions},
            {"max_neighbors", m_maxNeighbors}
        };
    }
    
    void deserialize(const json& j) override {
        if (j.contains("name")) m_name = j["name"];
        if (j.contains("birth")) m_birthConditions = j["birth"].get<std::vector<int>>();
        if (j.contains("survive")) m_surviveConditions = j["survive"].get<std::vector<int>>();
        if (j.contains("max_neighbors")) m_maxNeighbors = j["max_neighbors"];
    }
};

// Менеджер конфигурации
class ConfigManager {
private:
    std::map<std::string, std::unique_ptr<IConfigurable>> m_configurables;
    json m_extraSettings;
    
public:
    template<typename T>
    void registerConfigurable(const std::string& key, std::unique_ptr<T> obj) {
        static_assert(std::is_base_of<IConfigurable, T>::value, "T must inherit from IConfigurable");
        m_configurables[key] = std::move(obj);
    }
    
    void saveToFile(const std::string& filename) {
        json j;
        
        // Сохраняем все зарегистрированные объекты
        for (const auto& [key, obj] : m_configurables) {
            j["objects"][key] = obj->serialize();
        }
        
        // Сохраняем дополнительные настройки
        j["settings"] = m_extraSettings;
        
        std::ofstream file(filename);
        if (file.is_open()) {
            file << j.dump(4);
        }
    }
    
    void loadFromFile(const std::string& filename) {
        std::ifstream file(filename);
        if (!file.is_open()) return;
        
        json j = json::parse(file);
        
        // Загружаем объекты
        if (j.contains("objects")) {
            for (const auto& [key, objJson] : j["objects"].items()) {
                auto it = m_configurables.find(key);
                if (it != m_configurables.end()) {
                    it->second->deserialize(objJson);
                }
            }
        }
        
        // Загружаем дополнительные настройки
        if (j.contains("settings")) {
            m_extraSettings = j["settings"];
        }
    }
    
    json& getExtraSettings() { return m_extraSettings; }
};
```

### 6. **Полезные утилиты для JSON**

```cpp
// Конвертация TGUI Color в JSON
json colorToJson(const tgui::Color& color) {
    return {color.red, color.green, color.blue, color.alpha};
}

// Конвертация JSON в TGUI Color
tgui::Color colorFromJson(const json& j) {
    if (j.is_array() && j.size() >= 3) {
        return tgui::Color(
            j[0].get<uint8_t>(),
            j[1].get<uint8_t>(),
            j[2].get<uint8_t>(),
            j.size() >= 4 ? j[3].get<uint8_t>() : 255
        );
    }
    return tgui::Color::White;
}

// Глубокое слияние JSON объектов (полезно для мержа с дефолтными настройками)
json mergeJson(const json& base, const json& override) {
    json result = base;
    
    for (auto it = override.begin(); it != override.end(); ++it) {
        if (it.value().is_object() && result.contains(it.key()) && result[it.key()].is_object()) {
            result[it.key()] = mergeJson(result[it.key()], it.value());
        } else {
            result[it.key()] = it.value();
        }
    }
    
    return result;
}

// Пример использования с дефолтными настройками
json defaultConfig = {
    {"speed", 60},
    {"grid", true},
    {"colors", {
        {"bg", {255, 255, 255}},
        {"fg", {0, 0, 0}}
    }}
};

json userConfig = {
    {"speed", 120},
    {"colors", {
        {"bg", {0, 0, 0}}
    }}
};

json finalConfig = mergeJson(defaultConfig, userConfig);
// Результат: speed=120, grid=true, colors.bg={0,0,0}, colors.fg={0,0,0}
```
